                    },
                    {
                        title: '背景2',

                    },
                    {
                        title: '背景3',

                    },
                    {
                        title: '背景4',
                        src: "https://www.zpzpup.com/assets/image/example/bg10.jpeg"
                    },
                    {
                        title: '背景5',
                        src: "https://www.zpzpup.com/assets/image/example/bg7.jpeg"
                    }
                ],
                positionList: [
                    {
                        width: '172px',
                        height: '500px',
                        top: '-20px',
                        left: '60px',
                        right: '',
                        bottom: '',
                    },
                    {
                        width: '172px',
                        height: '500px',
                        top: '',
                        left: '300px',
                        right: '',
                        bottom: '100px',
                    },
                    {
                        width: '172px',
                        height: '500px',
                        top: '',
                        left: '-150px',
                        right: '',
                        bottom: '60px',
                    },
                    {
                        width: '172px',
                        height: '500px',
                        top: '-130px',
                        left: '',
                        right: '140px',
                        bottom: '',
                    },
                    {
                        width: '172px',
                        height: '500px',
                        top: '',
                        left: '',
                        right: '-50px',
                        bottom: '10px',
                    }
                ],
                flag: true
            }
        },
        mounted () {
            this.start()
            setInterval(() => {
                this.flag = false
                this.start()
            }, 6000)
        },
        methods: {
            start () {
                this.flag = true
                requestAnimationFrame(() => {
                    this.positionList = this.shuffle(this.positionList)
                })
            },
            shuffle (arr) {
                let i = arr.length;
                while (i) {
                    let j = Math.floor(Math.random() * i--);
                    [arr[j], arr[i]] = [arr[i], arr[j]];
                }
                return arr;
            }
        }
    }
